python train.py --data data/card_kpts.yaml --cfg models/hub/yolov5s6_4kpts.yaml --weights weights/yolov5s6_pose_960.pt --hyp data/hyp.scratch.yaml --img 960 --batch-size 16 --epochs 150 --kpt-label


step 1
python train.py --data data/card_kpts.yaml --cfg models/hub/yolov5s6_4kpts.yaml --weights weights/yolov5s6_pose_960.pt --hyp data/hyp.scratch.yaml --img 960 --batch-size 16 --epochs 50 --name step1_960_sigma_0.5 --kpt-label

step 2
change sigmas in loss.py to torch.tensor([0.05, 0.05, 0.05, 0.05]
and in data/card_coco_kpts.yaml to sigmas: [0.5, 0.5, 0.5, 0.5]
python train.py --data data/card_coco_kpts.yaml --cfg models/hub/yolov5s6_4kpts.yaml --weights runs/train/step1_960_sigma_0.5/weights/best.pt --hyp data/hyp.scratch.yaml --img-size 960 --batch-size 16 --epochs 100 --name step2_960_sigma_0.05 --kpt-label

step 3
change the sigma to 0.01
change hyp.scratch.yaml to hyp.finetune.yaml and update:
lr0: 0.001
lrf: 0.01
box: 0.10
kpt: 2.0
cls: 0.5
python train.py --data data/card_coco_kpts.yaml --cfg models/hub/yolov5s6_4kpts.yaml --weights runs/train/step2_sigma_0.05/weights/best.p --hyp data/hyp.finetune.yaml --img-size 960 --batch-size 16 --epochs 100 --name step3_960_sigma_0.01 --kpt-label
